Enter the closure

nested functions have access to the variables in the scope of the enclosing scope even after the enclosing function has terminated

enter the module

  • Eine umschliessende Funktion
  • private eigenschaften in diesem Scope
  • ein Object mit methoden ist public interface

enter the module


function module(inject, params, here){
  var some = 39, private = 1, props = 2;

  var theInterface = {};

  function imPrivilegedMathMagic( num ){
    return (some + num * props ) - private;
  }

  theInterface.mathMagic = imPrivilegedMathMagic;

  return theInterface;
}

enter the module as IIFE


(function (global, $){
  var some = 39, private = 1, props = 2;

  var theInterface = {};

  function imPrivilegedMathMagic( num ){
    return (some + num * props ) - private;
  }

  theInterface.mathMagic = imPrivilegedMathMagic;

  return theInterface;
})(window, jQuery)
wichtig hier: nichts bleibt im globalen scope übrig

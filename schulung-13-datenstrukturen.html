enter the module

  • An enclosing function

enter the module


function module(inject, params, here){
  var some = 39, private = 1, props = 2;

  var theInterface = {};

  function imPrivilegedMathMagic( num ){
    return (some + num * props ) - private;
  }

  theInterface.mathMagic = imPrivilegedMathMagic;

  return theInterface;
}
